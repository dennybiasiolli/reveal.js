### Install Vuex

```html
<script src="/path/to/vue.js"></script>
<script src="/path/to/vuex.js"></script>
```

or

```sh
npm install --save vuex
# or
yarn add vuex  # https://yarnpkg.com/
# or
npx @vue/cli add vuex  # https://cli.vuejs.org/
```

```js
import Vue from 'vue';
import Vuex from 'vuex';

Vue.use(Vuex);
```
